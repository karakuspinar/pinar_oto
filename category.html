document.addEventListener("DOMContentLoaded", async function(){ const params = new URLSearchParams(window.location.search); const kategori = params.get("cat"); document.getElementById("baslik").innerText = kategori || "Tüm Ürünler"; const brand = localStorage.getItem("carBrand"); const model = localStorage.getItem("carModel"); const year = localStorage.getItem("carYear"); const res = await fetch("products.json"); const products = await res.json(); let html=""; products.forEach(p=>{ // kategori kontrol if(kategori && p.kategori !== kategori) return; // araç filtresi yoksa direkt göster if(!brand){ html += kart(p); return; } // araç eşleşme if( p.brand === brand && p.model === model && p.year === year ){ html += kart(p); } }); document.getElementById("urunler").innerHTML = html; }); function kart(p){ return `

${p.ad}

${p.fiyat} TL

`; }
